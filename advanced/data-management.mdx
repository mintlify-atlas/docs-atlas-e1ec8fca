## Overview

Cotizador Online OSPADEP uses localStorage for client-side data persistence. The system manages healthcare plan prices, custom configurations, user sessions, and audit logs.

## Data Storage Architecture

### localStorage Keys

The application uses the following localStorage keys:

| Key | Description | Data Type |
|-----|-------------|----------|
| `ospadep_custom_prices` | Custom price modifications | JSON Object |
| `ospadep_custom_plans` | Custom plan configurations | JSON Object |
| `ospadep_users` | User accounts database | JSON Object |
| `ospadep_active_session` | Current user session | JSON Object |
| `ospadep_audit_log` | Activity and audit logs | JSON Array |
| `ospadep_last_sync` | Last synchronization timestamp | String |

## Custom Price Storage

### Saving Custom Prices

From `script.js`:

```javascript
function saveCustomPrices() {
  try {
    localStorage.setItem('ospadep_custom_prices', 
      JSON.stringify(planesData));
    
    // Log the save operation
    if (window.UserManager) {
      window.UserManager.logActivity('PRICES_SAVED', {
        timestamp: new Date().toISOString(),
        dataSize: JSON.stringify(planesData).length
      });
    }
  } catch (error) {
    console.error('Error saving custom prices:', error);
    alert('Error al guardar precios personalizados');
  }
}
```

### Loading Custom Prices

```javascript
function loadCustomPrices() {
  try {
    const saved = localStorage.getItem('ospadep_custom_prices');
    
    if (saved) {
      const customPrices = JSON.parse(saved);
      
      // Merge with base prices
      Object.assign(planesData, customPrices);
      
      console.log('Custom prices loaded successfully');
      return true;
    }
    
    return false;
  } catch (error) {
    console.error('Error loading custom prices:', error);
    return false;
  }
}
```

<Note>
  Custom prices override base prices from `precios-data.json`. Always load base prices first, then apply custom prices.
</Note>

### Loading Users

```javascript
loadUsers() {
  const defaultUsers = {
    'ospadep_admin_2024': {
      // SuperAdmin default user
    }
  };

  const saved = localStorage.getItem('ospadep_users');
  if (saved) {
    try {
      const users = JSON.parse(saved);
      // Ensure SuperAdmin always exists
      return { ...defaultUsers, ...users };
    } catch (error) {
      console.error('Error loading users:', error);
      return defaultUsers;
    }
  }
  
  // Save default users
  this.saveUsers(defaultUsers);
  return defaultUsers;
}
```

```javascript
saveToAuditLog(entry) {
  let auditLog = JSON.parse(
    localStorage.getItem('ospadep_audit_log') || '[]');
  
  auditLog.push(entry);
  
  // Keep only the most recent records
  if (auditLog.length > APP_CONFIG.MAX_AUDIT_RECORDS) {
    auditLog = auditLog.slice(-APP_CONFIG.MAX_AUDIT_RECORDS);
  }
  
  localStorage.setItem('ospadep_audit_log', 
    JSON.stringify(auditLog));
}
```

### Retrieving Audit Logs

```javascript
getAuditLog(filters = {}) {
  const auditLog = JSON.parse(
    localStorage.getItem('ospadep_audit_log') || '[]');
  
  let filteredLog = auditLog;
  
  // Apply filters
  if (filters.userId) {
    filteredLog = filteredLog.filter(
      entry => entry.userId === filters.userId);
  }
  
  if (filters.action) {
    filteredLog = filteredLog.filter(
      entry => entry.action === filters.action);
  }
  
  if (filters.dateFrom) {
    filteredLog = filteredLog.filter(
      entry => new Date(entry.timestamp) >= new Date(filters.dateFrom));
  }
  
  // Sort by date descending
  return filteredLog.sort(
    (a, b) => new Date(b.timestamp) - new Date(a.timestamp));
}
```

## Backup and Restore

### Complete Data Backup

```javascript
function backupAllData() {
  const backup = {
    exportDate: new Date().toISOString(),
    version: '2026-02',
    data: {
      prices: localStorage.getItem('ospadep_custom_prices'),
      plans: localStorage.getItem('ospadep_custom_plans'),
      users: localStorage.getItem('ospadep_users'),
      auditLog: localStorage.getItem('ospadep_audit_log')
    }
  };
  
  return backup;
}
```

### Export Data to File

```javascript
function exportBackup() {
  const backup = backupAllData();
  const dataStr = JSON.stringify(backup, null, 2);
  const dataBlob = new Blob([dataStr], {type: 'application/json'});
  
  const link = document.createElement('a');
  link.href = URL.createObjectURL(dataBlob);
  link.download = `ospadep_backup_${new Date().toISOString().slice(0, 10)}.json`;
  link.click();
  
  // Log export
  if (window.UserManager) {
    window.UserManager.logActivity('DATA_EXPORT', {
      dataSize: dataStr.length,
      timestamp: new Date().toISOString()
    });
  }
}
```

### Restore from Backup

```javascript
function restoreFromBackup(backupData) {
  try {
    // Validate backup structure
    if (!backupData.data || !backupData.version) {
      throw new Error('Invalid backup format');
    }
    
    // Restore each component
    if (backupData.data.prices) {
      localStorage.setItem('ospadep_custom_prices', 
        backupData.data.prices);
    }
    
    if (backupData.data.plans) {
      localStorage.setItem('ospadep_custom_plans', 
        backupData.data.plans);
    }
    
    if (backupData.data.users) {
      localStorage.setItem('ospadep_users', 
        backupData.data.users);
    }
    
    if (backupData.data.auditLog) {
      localStorage.setItem('ospadep_audit_log', 
        backupData.data.auditLog);
    }
    
    // Log restore
    if (window.UserManager) {
      window.UserManager.logActivity('DATA_RESTORE', {
        backupDate: backupData.exportDate,
        timestamp: new Date().toISOString()
      });
    }
    
    alert('Backup restaurado exitosamente');
    location.reload();
    
  } catch (error) {
    console.error('Error restoring backup:', error);
    alert('Error al restaurar backup: ' + error.message);
  }
}
```

### Import from File

```javascript
function importBackup() {
  const input = document.createElement('input');
  input.type = 'file';
  input.accept = '.json';
  
  input.onchange = (e) => {
    const file = e.target.files[0];
    const reader = new FileReader();
    
    reader.onload = (event) => {
      try {
        const backup = JSON.parse(event.target.result);
        restoreFromBackup(backup);
      } catch (error) {
        alert('Error al leer archivo: ' + error.message);
      }
    };
    
    reader.readAsText(file);
  };
  
  input.click();
}
```

## Storage Limits

### localStorage Capacity

```javascript
function checkStorageCapacity() {
  let total = 0;
  
  for (let key in localStorage) {
    if (localStorage.hasOwnProperty(key)) {
      total += localStorage[key].length + key.length;
    }
  }
  
  const totalKB = (total / 1024).toFixed(2);
  const limit = 5120; // 5MB typical limit
  const percentUsed = ((total / 1024) / limit * 100).toFixed(2);
  
  return {
    used: totalKB + ' KB',
    limit: limit + ' KB',
    percentUsed: percentUsed + '%'
  };
}
```

<Warning>
  localStorage typically has a 5-10MB limit per domain. Monitor storage usage and implement cleanup strategies for large audit logs.
</Warning>

## Data Cleanup

### Clear Old Audit Logs

```javascript
function cleanupOldAuditLogs(daysToKeep = 30) {
  const auditLog = JSON.parse(
    localStorage.getItem('ospadep_audit_log') || '[]');
  
  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - daysToKeep);
  
  const filtered = auditLog.filter(entry => 
    new Date(entry.timestamp) > cutoffDate);
  
  localStorage.setItem('ospadep_audit_log', 
    JSON.stringify(filtered));
  
  return auditLog.length - filtered.length; // removed count
}
```
